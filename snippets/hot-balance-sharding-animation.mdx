export const HotBalanceShardingAnimation = () => {
  const [playing, setPlaying] = useState(true)
  const [step, setStep] = useState(0)
  const [reduced, setReduced] = useState(false)
  const shards = ["0", "1", "2", "3"]

  const STEP_MS = 3000

  useEffect(() => {
    const mq = window.matchMedia("(prefers-reduced-motion: reduce)")
    const onMotionChange = () => setReduced(mq.matches)
    onMotionChange()
    mq.addEventListener("change", onMotionChange)
    return () => mq.removeEventListener("change", onMotionChange)
  }, [])

  useEffect(() => {
    if (reduced || !playing) {
      return undefined
    }

    let cancelled = false
    let timeoutId

    const tick = () => {
      timeoutId = window.setTimeout(() => {
        if (cancelled) {
          return
        }
        setStep((current) => (current + 1) % 4)
        tick()
      }, STEP_MS)
    }

    tick()

    return () => {
      cancelled = true
      window.clearTimeout(timeoutId)
    }
  }, [playing, reduced])

  const currentStep = reduced ? 3 : step
  const captions = [
    "All traffic targets one hot settlement balance.",
    "The balance splits into parallel shard lanes.",
    "A router distributes each transaction to one shard.",
    "Shards process concurrently with less lock contention.",
  ]

  return (
    <figure className="hb-anim not-prose" aria-label="Balance sharding spreads traffic across parallel shard balances">
      <div className="hb-anim-stage">
        <button type="button" className="hb-anim-toggle" onClick={() => setPlaying((current) => !current)} aria-pressed={playing}>
          {playing ? "Pause" : "Play"}
        </button>
        <svg viewBox="0 0 720 220" className="hb-anim-svg" role="img" aria-hidden="true">
          <defs>
            <marker id="hb-shard-arrowhead" markerWidth="8" markerHeight="8" refX="6" refY="3" orient="auto">
              <path d="M0,0 L6,3 L0,6 Z" className="hb-anim-arrowhead" />
            </marker>
          </defs>
          <g className="hb-anim-layer" opacity={currentStep === 0 ? 1 : 0}>
            <g transform="translate(300, 88)">
              <rect x={0} y={0} width={120} height={40} rx={8} className="hb-anim-node hb-anim-node--active hb-anim-node--accent" />
              <text x={60} y={24} textAnchor="middle" className="hb-anim-label">@Settlement</text>
            </g>
            <circle cx={200} cy={60} r={5} className="hb-anim-tx hb-anim-tx--active" />
            <circle cx={220} cy={80} r={5} className="hb-anim-tx hb-anim-tx--active" />
            <circle cx={210} cy={100} r={5} className="hb-anim-tx hb-anim-tx--active" />
            <circle cx={230} cy={120} r={5} className="hb-anim-tx hb-anim-tx--active" />
            <line x1={235} y1={108} x2={300} y2={108} className="hb-anim-arrow hb-anim-arrow--active" strokeDasharray="5 4" markerEnd="url(#hb-shard-arrowhead)" />
          </g>
          <g className="hb-anim-layer" opacity={currentStep >= 1 ? 1 : 0}>
            {shards.map((shard, index) => (
              <g key={shard} transform={`translate(${80 + index * 140}, 88)`}>
                <rect x={0} y={0} width={108} height={40} rx={8} className={`hb-anim-node${currentStep >= 3 ? " hb-anim-node--active hb-anim-node--accent" : ""}`} />
                <text x={54} y={24} textAnchor="middle" className="hb-anim-label">{`@Settlement-${shard}`}</text>
              </g>
            ))}
            <text x={620} y={112} className="hb-anim-annotation">+16</text>
          </g>
          <g className="hb-anim-layer" opacity={currentStep === 2 ? 1 : 0}>
            <g transform="translate(300, 24)">
              <rect x={0} y={0} width={80} height={32} rx={8} className="hb-anim-node hb-anim-node--active hb-anim-node--accent" />
              <text x={40} y={20} textAnchor="middle" className="hb-anim-label">Router</text>
            </g>
            <circle cx={340} cy={72} r={5} className="hb-anim-tx hb-anim-tx--active" />
            <line x1={340} y1={56} x2={134} y2={88} className="hb-anim-arrow hb-anim-arrow--active" strokeDasharray="5 4" markerEnd="url(#hb-shard-arrowhead)" />
            <line x1={340} y1={56} x2={274} y2={88} className="hb-anim-arrow hb-anim-arrow--active" strokeDasharray="5 4" markerEnd="url(#hb-shard-arrowhead)" />
            <line x1={340} y1={56} x2={414} y2={88} className="hb-anim-arrow hb-anim-arrow--active" strokeDasharray="5 4" markerEnd="url(#hb-shard-arrowhead)" />
            <line x1={340} y1={56} x2={554} y2={88} className="hb-anim-arrow hb-anim-arrow--active" strokeDasharray="5 4" markerEnd="url(#hb-shard-arrowhead)" />
          </g>
          <g className="hb-anim-layer" opacity={currentStep === 3 ? 1 : 0}>
            <line x1={40} y1={140} x2={134} y2={128} className="hb-anim-arrow hb-anim-arrow--active" markerEnd="url(#hb-shard-arrowhead)" />
            <line x1={40} y1={160} x2={274} y2={128} className="hb-anim-arrow hb-anim-arrow--active" markerEnd="url(#hb-shard-arrowhead)" />
            <line x1={40} y1={180} x2={414} y2={128} className="hb-anim-arrow hb-anim-arrow--active" markerEnd="url(#hb-shard-arrowhead)" />
            <line x1={40} y1={200} x2={554} y2={128} className="hb-anim-arrow hb-anim-arrow--active" markerEnd="url(#hb-shard-arrowhead)" />
            <text x={40} y={170} className="hb-anim-annotation hb-anim-annotation--accent">parallel lanes</text>
          </g>
        </svg>
      </div>
      <figcaption className="hb-anim-caption">{captions[currentStep]}</figcaption>
    </figure>
  )
}

<HotBalanceShardingAnimation />

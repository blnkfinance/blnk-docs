export const HotBalanceHotLaneAnimation = () => {
  const [playing, setPlaying] = useState(true)
  const [step, setStep] = useState(0)
  const [reduced, setReduced] = useState(false)

  const STEP_MS = 3000

  useEffect(() => {
    const mq = window.matchMedia("(prefers-reduced-motion: reduce)")
    const onMotionChange = () => setReduced(mq.matches)
    onMotionChange()
    mq.addEventListener("change", onMotionChange)
    return () => mq.removeEventListener("change", onMotionChange)
  }, [])

  useEffect(() => {
    if (reduced || !playing) {
      return undefined
    }

    let cancelled = false
    let timeoutId

    const tick = () => {
      timeoutId = window.setTimeout(() => {
        if (cancelled) {
          return
        }
        setStep((current) => (current + 1) % 4)
        tick()
      }, STEP_MS)
    }

    tick()

    return () => {
      cancelled = true
      window.clearTimeout(timeoutId)
    }
  }, [playing, reduced])

  const currentStep = reduced ? 3 : step
  const captions = [
    "Mixed traffic flows through the main transaction queue.",
    "One balance pair keeps colliding under heavy concurrent load.",
    "Blnk routes that pair into a dedicated hot lane queue.",
    "The main queue keeps processing other traffic unblocked.",
  ]

  return (
    <figure className="hb-anim not-prose" aria-label="Hot-lane routing isolates a contested balance pair into a dedicated queue">
      <div className="hb-anim-stage">
        <button type="button" className="hb-anim-toggle" onClick={() => setPlaying((current) => !current)} aria-pressed={playing}>
          {playing ? "Pause" : "Play"}
        </button>
        <svg viewBox="0 0 720 220" className="hb-anim-svg" role="img" aria-hidden="true">
          <defs>
            <marker id="hb-hotlane-arrowhead" markerWidth="8" markerHeight="8" refX="6" refY="3" orient="auto">
              <path d="M0,0 L6,3 L0,6 Z" className="hb-anim-arrowhead" />
            </marker>
          </defs>
          <g transform="translate(80, 40)">
            <rect x={0} y={0} width={200} height={72} rx={8} className={`hb-anim-queue${currentStep === 0 || currentStep === 3 ? " hb-anim-queue--active" : ""}`} />
            <text x={100} y={16} textAnchor="middle" className="hb-anim-label hb-anim-label--small">Queue</text>
            {Array.from({ length: 4 }).map((_, index) => (
              <rect key={index} x={14 + index * 4} y={28 + index * 6} width={172} height={8} rx={4} className={`hb-anim-queue-item${currentStep === 0 || currentStep === 3 ? " hb-anim-queue-item--active" : ""}`} />
            ))}
          </g>
          <g transform="translate(80, 132)">
            <rect x={0} y={0} width={200} height={72} rx={8} className={`hb-anim-queue${currentStep >= 2 ? " hb-anim-queue--active" : ""}`} />
            <text x={100} y={16} textAnchor="middle" className="hb-anim-label hb-anim-label--small">Hot lane</text>
            {Array.from({ length: 3 }).map((_, index) => (
              <rect
                key={index}
                x={14 + index * 4}
                y={28 + index * 6}
                width={172}
                height={8}
                rx={4}
                opacity={currentStep >= 2 || index === 0 ? 1 : 0}
                className={`hb-anim-queue-item${currentStep >= 2 ? " hb-anim-queue-item--active" : ""}`}
              />
            ))}
          </g>
          <g transform="translate(360, 52)">
            <rect x={0} y={0} width={96} height={40} rx={8} className={`hb-anim-node${currentStep === 1 ? " hb-anim-node--active hb-anim-node--accent" : ""}`} />
            <text x={48} y={24} textAnchor="middle" className="hb-anim-label">Balance A</text>
          </g>
          <g transform="translate(360, 144)">
            <rect x={0} y={0} width={96} height={40} rx={8} className={`hb-anim-node${currentStep === 1 ? " hb-anim-node--active hb-anim-node--accent" : ""}`} />
            <text x={48} y={24} textAnchor="middle" className="hb-anim-label">Balance B</text>
          </g>
          <g transform="translate(560, 52)">
            <rect x={0} y={0} width={96} height={40} rx={8} className={`hb-anim-node${currentStep === 3 ? " hb-anim-node--active" : ""}`} />
            <text x={48} y={24} textAnchor="middle" className="hb-anim-label">Balance C</text>
          </g>
          <g transform="translate(560, 144)">
            <rect x={0} y={0} width={96} height={40} rx={8} className={`hb-anim-node${currentStep === 3 ? " hb-anim-node--active" : ""}`} />
            <text x={48} y={24} textAnchor="middle" className="hb-anim-label">Balance D</text>
          </g>
          <line x1={280} y1={76} x2={360} y2={72} className={`hb-anim-arrow${currentStep === 0 ? " hb-anim-arrow--active" : ""}`} strokeDasharray="5 4" markerEnd="url(#hb-hotlane-arrowhead)" />
          <line x1={280} y1={76} x2={560} y2={72} className={`hb-anim-arrow${currentStep === 3 ? " hb-anim-arrow--active" : ""}`} strokeDasharray="5 4" markerEnd="url(#hb-hotlane-arrowhead)" />
          <line x1={280} y1={168} x2={360} y2={164} className={`hb-anim-arrow${currentStep >= 2 ? " hb-anim-arrow--active" : ""}`} markerEnd="url(#hb-hotlane-arrowhead)" />
          <line x1={456} y1={72} x2={456} y2={164} className={`hb-anim-arrow${currentStep === 1 ? " hb-anim-arrow--active" : ""}`} strokeDasharray="5 4" markerEnd="url(#hb-hotlane-arrowhead)" />
          <line x1={456} y1={164} x2={280} y2={168} className={`hb-anim-arrow${currentStep === 2 ? " hb-anim-arrow--active" : ""}`} markerEnd="url(#hb-hotlane-arrowhead)" />
          <text x={470} y={118} className="hb-anim-annotation hb-anim-annotation--accent" opacity={currentStep === 1 ? 1 : 0}>contention</text>
          <text x={300} y={36} className="hb-anim-annotation" opacity={currentStep === 3 ? 1 : 0}>other pairs continue</text>
        </svg>
      </div>
      <figcaption className="hb-anim-caption">{captions[currentStep]}</figcaption>
    </figure>
  )
}

<HotBalanceHotLaneAnimation />
